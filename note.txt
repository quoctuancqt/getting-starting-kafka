CREATE STREAM MQTT_DEVICE_TELEMETRIES(temperature DOUBLE,humidity BIGINT) WITH (kafka_topic ='MQTT_DEVICE_TELEMETRIES', value_format ='json');

CREATE STREAM DEVICE_TELEMETRIES WITH (VALUE_FORMAT='AVRO') AS
	SELECT 
    	temperature as "temperature",
    	humidity as "humidity"
  	FROM  MQTT_DEVICE_TELEMETRIES
EMIT CHANGES;
